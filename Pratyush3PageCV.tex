	\cventry{2018 - Current}{Fast online execution of model predictive control

		neural networks}{}{}{}{PhD thesis project, University of California, 
		Santa Barbara.}
	\cvlistitem{Developing algorithms to replace real-time optimization with

	problems.}
	\cvlistitem{Engineered the offline design of NNs using solutions of MPC 
	optimization problems for use in real-time instead of online optimization 
	solvers.}
	\cvlistitem{Invented a structured NN architecture for offset-free setpoint

	offline network design step and online computational speed-ups of four 
	orders of magnitude by NNs over online quadratic programming solvers.}
	\cventry{2020 -
		Current}{Hybrid grey-box and data-driven system identification with 
		machine learning}{}{}{}{PhD thesis project, University of California, 
		Santa Barbara.}
	\cvlistitem{Developed methods to augment traditional first principles 
	grey-box models with NNs for use in model-based feedback control 
	applications.}
	\cvlistitem{Investigating data requirements of the system identification 
	approach; developing methods for optimization over NN models; conducting 
	simulation studies to demonstrate the industrial applicability of the 
	approach.}
	\cventry{2020 - Current}{Modeling proportional-integral controllers in 
	model predictive control}{}{}{}{Research project, University of California, 
	Santa Barbara.}
	\cvlistitem{Examined performances of MPC controllers after including 
	dynamic models of proportional-integral controllers in MPC optimization 
	problems. Attained noticeable performance benefits of 8-50 $\%$ over the 
	traditional approach of not modeling PI controllers in MPC formulations.}
	\cventry{2018}{Reinforcement learning for chemical process 
	control}{}{}{}{PhD thesis project, University of California, Santa Barbara.}
	\cvlistitem{Investigated reinforcement learning (RL) algorithms such as 
	Q-learning and policy gradient for data-driven and model-free control of 
	chemical process systems. Conducted simulation studies and compared sample 
	efficiency of RL with the model-based controller design approach.}
